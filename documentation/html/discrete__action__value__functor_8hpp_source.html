121  template<typename StateType, typename ActionType>
122  double DiscreteActionValue<StateType, ActionType>::
123  Get(const StateType& state, const ActionType& action) const {
124  if (value_.count(state) == 0)
125  return kInvalidValue;
126 
127  if (value_.at(state).count(action) == 0)
128  return kInvalidValue;
129 
130  return value_.at(state).at(action);
131  }

133  template<typename StateType, typename ActionType>
134  void DiscreteActionValue<StateType, ActionType>::
135  Set(const StateType& state, const ActionType& action, double value) {
136  if (value_.count(state) > 0) {
137  if (value_.at(state).count(action) > 0)
138  // Table contains this pair.
139  value_.at(state).at(action) = value;
140  else
141  // Table contains state but not action.
142  value_.at(state).insert({action, value});
143  } else {
144  // Table does not contain this state.
145  value_.insert({state, std::unordered_map<
146  ActionType, double, typename ActionType::Hash>({{action, value}})});
147  }
148  }

156  template<typename StateType, typename ActionType>
157  void DiscreteActionValue<StateType, ActionType>::
158  Initialize(const DiscreteEnvironment<StateType, ActionType>& environment) {
159  value_.clear();
160 
161  // Get a list of all states.
162  std::vector<StateType> states;
163  environment.States(states);
164 
165  for (const auto& state : states) {
166  // Get a list of all feasible actions in this state.
167  std::vector<ActionType> actions;
168  environment.Actions(state, actions);
169 
170  for (const auto& action : actions)
171  Set(state, action, 0.0);
172  }
173  }
