95  // Runs Q Learning algorithm for the specified number of iterations.
96  // Solution is stored in the provided continuous value functor.
97  template<typename StateType, typename ActionType>
98  void ContinuousQLearning<StateType, ActionType>::Solve(
99  const ContinuousEnvironment<StateType, ActionType>& environment,
100  ContinuousActionValueFunctor& value) {
101  // Run for the specified number of iterations. Assume value functor
102  // has already been initialized.
103  for (size_t ii = 1; ii <= num_rollouts_; ii++)
104  UpdateValueFunction(environment);

107  // Estimate the value function for the current policy using Q Learning,
108  // from a single rollout.
109  template<typename StateType, typename ActionType>
110  void ContinuousQLearning<StateType, ActionType>::UpdateValueFunction(
111  const ContinousEnvironment<StateType, ActionType>& environment,
112  ContinuousActionValueFunctor& value) {
113  // Initialize the current state to the initial state.
114  StateType current_state = initial_state_;
115 
116  // Get the optimal action in this state.
117  ActionType current_action;
118  if (!value.OptimalAction(current_state, current_action))
119  LOG(WARNING) << "ContinuousQLearning: Could not find optimal action.";
120 
121  // Start a new ExperienceReplay dataset.
122  ExperienceReplay<StateType, ActionType> replay;
123 
124  // Simulate the rollout.
125  for (int ii = 0; (ii < rollout_length_) ||
126  (rollout_length_ < 0 && !environment.IsTerminal(current_state));
127  ii++) {
128  // Take the current action.
129  StateType next_state = current_state;
130  const double reward = environment.Simulate(next_state, current_action);
131 
132  // Get optimal next action.
133  ActionType next_action;
134  if (!value.OptimalAction(next_state, next_action))
135  LOG(WARNING) << "ContinuousQLearning: Could not find optimal action.";
136 
137  // Store this experience in the replay unit.
138  replay.Add(current_state, current_action, reward, next_state);
139 
140  // Replay experience and each time update the value function.
141  for (size_t jj = 0; jj < num_exp_replays_; jj++) {
142  StateType sample_state, sample_next_state;
143  ActionType sample_action;
144  double sample_reward;
145 
146  CHECK(replay.Sample(sample_state, sample_action,
147  sample_next_state, sample_reward));
148 
149  // Compute the Q-Learning target at the sample.
150  ActionType optimal_next_action;
151  if (!value.OptimalAction(sample_next_state, optimal_next_action))
152  LOG(WARNING) << "ContinuousQLearning: Could not find optimal action.";
153 
154  const double sample_value = value(sample_state, sample_action);
155  const double target =
156  sample_value + alpha_ * (sample_reward +
157  value(sample_next_state, optimal_next_action) - sample_value);
158 
159  // Update.
160  value.Update(sample_state, sample_action, target, learning_rate_);
161  }
162  }
163  }
