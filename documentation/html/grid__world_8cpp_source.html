56  // Implement pure virtual method from Environment, but leave it virtual
57  // so that a derived class can override it (for example, adding some
58  // randomness, or other biases).
59  double GridWorld::Simulate(GridState& state, const GridAction& action) const {
60  // Handle all edge cases.
61  if (state.ii_ == 0 && action == GridAction::Direction::UP)
62  return kInvalidReward;
63  else if (state.ii_ == nrows_ - 1 && action == GridAction::Direction::DOWN)
64  return kInvalidReward;
65  else if (state.jj_ == 0 && action == GridAction::Direction::LEFT)
66  return kInvalidReward;
67  else if (state.jj_ == ncols_ - 1 && action == GridAction::Direction::RIGHT)
68  return kInvalidReward;
69 
70  // Definitely going to stay on the grid, so just parse action normally.
71  switch (action.direction_) {
72  case GridAction::Direction::UP :
73  state.ii_--;
74  break;
75  case GridAction::Direction::DOWN :
76  state.ii_++;
77  break;
78  case GridAction::Direction::LEFT :
79  state.jj_--;
80  break;
81  case GridAction::Direction::RIGHT :
82  state.jj_++;
83  break;
84 
85  default :
86  // Should never get here.
87  LOG(ERROR) << "GridWorld: Error! Parsing unknown action.";
88  return kInvalidReward;
89  }
90 
91  // Return whether or not we are in the goal state. Penalize not being in
92  // the goal state in order to get short paths.
93  const double reward = (state == goal_) ? 1.0 : -1.0;
94  return reward;
95  }

97  // Implement pure virtual method to return whether or not an action is
98  // valid in a given state.
99  bool GridWorld::IsValid(const GridState& state,
100  const GridAction& action) const {
101  switch (action.direction_) {
102  case GridAction::Direction::UP :
103  if (state.ii_ > 0)
104  return true;
105  break;
106  case GridAction::Direction::DOWN :
107  if (state.ii_ < nrows_ - 1)
108  return true;
109  break;
110  case GridAction::Direction::LEFT :
111  if (state.jj_ > 0)
112  return true;
113  break;
114  case GridAction::Direction::RIGHT :
115  if (state.jj_ < ncols_ - 1)
116  return true;
117  break;
118  default :
119  // Should never get here.
120  LOG(ERROR) << "GridWorld: Error. Invalid action direction.";
121  return false;
122  }
123 
124  return false;
125  }

135  states.clear();
136 
137  for (size_t ii = 0; ii < nrows_; ii++)
138  for (size_t jj = 0; jj < ncols_; jj++)
139  states.push_back(GridState(ii, jj));

142  void GridWorld::Actions(const GridState& state,
143  std::vector<GridAction>& actions) const {
144  actions.clear();
145 
146  // Catch invalid states.
147  CHECK_LT(state.ii_, nrows_);
148  CHECK_LT(state.jj_, ncols_);
149 
150  // Create a list of possible next states.
151  const std::vector<GridAction::Direction> possibilities =
152  {GridAction::Direction::UP, GridAction::Direction::DOWN,
153  GridAction::Direction::LEFT, GridAction::Direction::RIGHT};
154 
155  for (const auto& direction : possibilities) {
156  // Catch edge cases.
157  if (state.ii_ == 0 && direction == GridAction::Direction::UP)
158  continue;
159  if (state.ii_ == nrows_ - 1 && direction == GridAction::Direction::DOWN)
160  continue;
161  if (state.jj_ == 0 && direction == GridAction::Direction::LEFT)
162  continue;
163  if (state.jj_ == ncols_ - 1 && direction == GridAction::Direction::RIGHT)
164  continue;
165 
166  // If we get here, the direction must be valid.
167  actions.push_back(GridAction(direction));
168  }
169  }

171  // Visualize using OpenGL.
172  void GridWorld::Visualize() const {
173  glClear(GL_COLOR_BUFFER_BIT);
174  const GLfloat kEpsilon = 0.02;
175 
176  // Display each grid cell as a GL_QUAD, centered at the appropriate
177  // location, with a small 'epsilon' fudge factor between cells. Color
178  // the goal state red.
179  glBegin(GL_QUADS);
180  for (size_t ii = 0; ii < nrows_; ii++) {
181  for (size_t jj = 0; jj < ncols_; jj++) {
182  // Check for goal state.
183  if (ii == goal_.ii_ && jj == goal_.jj_)
184  glColor4f(0.8, 0.0, 0.2, 0.9);
185  else
186  glColor4f(0.9, 0.9, 0.9, 0.9);
187 
188  // Convert to top left x, y coords.
189  const GLfloat x = static_cast<GLfloat>(jj);
190  const GLfloat y = static_cast<GLfloat>(nrows_ - ii);
191 
192  // Bottom left, bottom right, top right, top left.
193  glVertex2f(x + kEpsilon, y - 1.0 + kEpsilon);
194  glVertex2f(x + 1.0 - kEpsilon, y - 1.0 + kEpsilon);
195  glVertex2f(x + 1.0 - kEpsilon, y - kEpsilon);
196  glVertex2f(x + kEpsilon, y - kEpsilon);
197  }
198  }
199  glEnd();
200  }
